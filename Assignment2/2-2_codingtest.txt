#include <string>
#include <vector>

using namespace std;

int solution(vector<vector<int>> board, int k) {
    int answer = 0;

    // 입력값 k에 대한 값 계산
    for (int i = 0; i < board.size(); i++) {
        for (int j = 0; j < board[i].size(); j++) {
            if (i + j > k) break;
            answer += board[i][j];
        }
    }
    return answer;
}
