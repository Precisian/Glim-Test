#include <string>
#include <vector>

using namespace std;

int solution(vector<vector<int>> dots) {
    int answer = 0;

    // 반복횟수 파악
    int repeat = dots.size();

    // 최대값이 존재하는 좌표 인덱스
    int _max_width = dots[0][0];
    int _min_width = dots[0][0];
    int _max_height = dots[0][1];
    int _min_height = dots[0][1];

    // 최대 너비 및 높이 좌표 확인
    for (int i = 0; i < repeat; i++) {
        if (_max_width < dots[i][0]) _max_width = dots[i][0];
        else if (_min_width > dots[i][0]) _min_width = dots[i][0];
        if (_max_height < dots[i][1]) _max_height = dots[i][1];
        else if (_min_height > dots[i][1]) _min_height = dots[i][1];
    }

    // 각 최대값에서 시작좌표값 차이 계산, 너비 및 높이 확인
    int width = abs(_max_width - _min_width);
    int height = abs(_max_height - _min_height);

    // 넓이 계산
    answer = width * height;

    return answer;
}
